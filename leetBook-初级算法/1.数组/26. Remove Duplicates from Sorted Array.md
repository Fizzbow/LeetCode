    var removeDuplicates = function (nums) {
      if (nums.length == 0) return 0;
      let f = 1,
        s = 1;
      while (f < nums.length) {
        if (nums[f - 1] !== nums[f]) {
          nums[s] = nums[f];
          ++s;
        }
        ++f;
      }
      return s;
